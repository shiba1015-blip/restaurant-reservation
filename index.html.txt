    <script>
        // 時間スロット生成
        function generateTimeSlots() {
            const timeSlotsContainer = document.getElementById('timeSlots');
            const times = [];
            
            // 10:00-20:30の30分刻み
            for (let hour = 10; hour <= 20; hour++) {
                for (let minute = 0; minute < 60; minute += 30) {
                    if (hour === 20 && minute > 30) break;
                    const timeString = `${hour.toString().padStart(2, '0')}:${minute.toString().padStart(2, '0')}`;
                    times.push(timeString);
                }
            }
            
            times.forEach(time => {
                const timeSlot = document.createElement('div');
                timeSlot.className = 'time-slot';
                timeSlot.textContent = time;
                timeSlot.addEventListener('click', () => selectTime(timeSlot, time));
                timeSlotsContainer.appendChild(timeSlot);
            });
        }
        
        // 時間選択
        function selectTime(element, time) {
            // 既存の選択をクリア
            document.querySelectorAll('.time-slot').forEach(slot => {
                slot.classList.remove('selected');
            });
            
            // 新しい選択を追加
            element.classList.add('selected');
            document.getElementById('selectedTime').value = time;
        }
        
        // 日付の最小値を設定（明後日から）
        function setMinDate() {
            const today = new Date();
            const dayAfterTomorrow = new Date(today);
            dayAfterTomorrow.setDate(today.getDate() + 2);
            
            const dateInput = document.getElementById('date');
            dateInput.min = dayAfterTomorrow.toISOString().split('T')[0];
        }
        
        // フォーム送信処理
        document.getElementById('reservationForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            // メールアドレス確認
            const email = document.getElementById('email').value;
            const emailConfirm = document.getElementById('emailConfirm').value;
            
            if (email !== emailConfirm) {
                alert('メールアドレスが一致しません。もう一度ご確認ください。');
                return;
            }
            
            // バリデーション
            const adults = parseInt(document.getElementById('adults').value) || 0;
            const children = parseInt(document.getElementById('children').value) || 0;
            const totalPeople = adults + children;
            
            if (totalPeople >= 9) {
                alert('9名以上のご予約はお電話にてお願いします。');
                return;
            }
            
            if (!document.getElementById('selectedTime').value) {
                alert('ご希望時間を選択してください。');
                return;
            }
            
            // チェックボックス確認
            const checks = ['check1', 'check2', 'check3'];
            for (let checkId of checks) {
                if (!document.getElementById(checkId).checked) {
                    alert('すべての注意事項にご同意いただく必要があります。');
                    return;
                }
            }
            
            // 予約データ収集
            const formData = {
                email: email,
                name: document.getElementById('name').value,
                furigana: document.getElementById('furigana').value,
                phone: document.getElementById('phone').value,
                date: document.getElementById('date').value,
                time: document.getElementById('selectedTime').value,
                adults: adults,
                children: children,
                totalPeople: totalPeople
            };
            
            console.log('予約データ:', formData);
            
            // 成功メッセージ表示
            document.getElementById('reservationForm').style.display = 'none';
            document.getElementById('successMessage').style.display = 'block';
            
            // 実際の運用では、ここでサーバーにデータを送信
            // sendReservationData(formData);
        });
        
        // 初期化
        document.addEventListener('DOMContentLoaded', function() {
            generateTimeSlots();
            setMinDate();
        });
    </script>
